CREATE TABLE PERSONNE(
	IM VARCHAR(10) PRIMARY KEY,
	nom VARCHAR(50),
	prenoms VARCHAR(50),
	datenais DATE,
	diplome VARCHAR(20),
	contact VARCHAR(10),
	status  VARCHAR(5),
	situation VARCHAR(20),
	idConjoint VARCHAR(11)
);
CREATE TABLE TARIF(
	num_tarif VARCHAR(3) PRIMARY KEY,
	diplome VARCHAR(20),
	categorie VARCHAR(50),
	montant INT
);

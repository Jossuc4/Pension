DROP TABLE IF EXISTS PERSONNE;
DROP TABLE IF EXISTS CONJOINT;
CREATE TABLE PERSONNE(
	IM VARCHAR(10) PRIMARY KEY,
	nom VARCHAR(50),
	prenoms VARCHAR(50),
	datenais DATE,
	diplome VARCHAR(20),
	contact VARCHAR(10),
	status  VARCHAR(5),
	situation VARCHAR(20),
	idConjoint VARCHAR(11) UNIQUE,
	FOREIGN KEY(idConjoint) REFERENCES CONJOINT(idConjoint)	
);
CREATE TABLE TARIF(
	num_tarif VARCHAR(3) PRIMARY KEY,
	diplome VARCHAR(20),
	categorie VARCHAR(50),
	montant INT
);
CREATE TABLE CONJOINT(
	idConjoint VARCHAR(11) UNIQUE,
	numPension VARCHAR(10),
	nomConjoint VARCHAR(50),
	prenomConjoint VARCHAR(50),
	montant INT,
	PRIMARY KEY(idConjoint,numPension)
);
CREATE TABLE PAYER(
	IM VARCHAR(10),
	datePayement DATE ,
	PRIMARY KEY(IM,datePayement)
);
